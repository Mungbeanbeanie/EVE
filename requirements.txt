# ─── EVE runtime dependencies ────────────────────────────────────────────────
# Lower bounds (>=) only; pin exact versions before a production deployment.

# LLM — provider-agnostic. LiteLLM gives one interface to Anthropic, OpenAI,
# Gemini, Ollama, etc. A provider SDK is needed only for a direct client in
# eve/llm/providers.py, and those imports are lazy, so missing SDKs are fine.
litellm>=1.40

# Config / env
python-dotenv>=1.0
pydantic-settings>=2.2

# Speech-to-text (local). portaudio is a SYSTEM package (PyAudio) — see README.
# faster-whisper runs Whisper via CTranslate2 (NO torch) — works on Intel macOS,
# CPU-friendly, and often faster than openai-whisper. Pass it a numpy PCM array.
faster-whisper>=1.0
webrtcvad>=2.0.10    # voice-activity / silence detection
# webrtcvad imports the legacy pkg_resources, which setuptools removed in v81.
# Pin below 81 so `import webrtcvad` works on Python 3.12+ venvs.
setuptools<81
pyaudio>=0.2.14      # raw mic capture + speaker playback

# Text-to-speech (local). Always-available offline voice + the runtime fallback.
pyttsx3>=2.90
# Text-to-speech (cloud, optional). Used only when ELEVENLABS_API_KEY is set; the
# import is lazy, so EVE runs fine on the local voice if this isn't installed.
elevenlabs>=2.0,<3

# Wake-word detection (local, no torch) — only needed for VOICE_INPUT=wake.
# Pretrained ONNX keyword models ("hey_jarvis", "alexa", …) run on CPU.
openwakeword>=0.6

# Memory — mem0 orchestrates embedding + vector search, over a FAISS index on disk
# and FastEmbed (ONNX) embeddings that run in-process.
mem0ai>=0.1.0
faiss-cpu>=1.8       # local vector index, persisted to disk
fastembed>=0.3       # in-process ONNX embeddings
groq>=1.0            # required when llm_provider=groq
numpy>=1.26

# Tools — Google (Gmail / Calendar / Drive) + Tavily web search
google-api-python-client>=2.120
google-auth>=2.29
google-auth-oauthlib>=1.2
requests>=2.31       # HTTP for the Tavily web-search adapter

# ─── Native desktop window (optional, macOS) ─────────────────────────────────
# Only needed for the always-on NATIVE window: `python main.py --window`.
# Without these, --window still works — it just opens the orb in your browser.
#   rumps     → menu-bar (status-bar) app host; runs Dock-less (LSUIElement).
#   pyobjc-*  → AppKit + WKWebView bindings used to host the web UI in a window.
#   pywebview → fallback plain-window backend when the menu-bar host is missing.
rumps>=0.4 ; sys_platform == "darwin"
pyobjc-framework-Cocoa>=10.0 ; sys_platform == "darwin"
pyobjc-framework-WebKit>=10.0 ; sys_platform == "darwin"
pywebview>=5.0 ; sys_platform == "darwin"

# ─── Dev / test ──────────────────────────────────────────────────────────────
pytest>=8.0

ruff>=0.4
